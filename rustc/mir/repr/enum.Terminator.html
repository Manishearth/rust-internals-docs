pub enum Terminator<'tcx> {
    Goto {
        target: BasicBlock,
    },
    If {
        cond: Operand<'tcx>,
        targets: (BasicBlock, BasicBlock),
    },
    Switch {
        discr: Lvalue<'tcx>,
        adt_def: AdtDef<'tcx>,
        targets: Vec<BasicBlock>,
    },
    SwitchInt {
        discr: Lvalue<'tcx>,
        switch_ty: Ty<'tcx>,
        values: Vec<ConstVal>,
        targets: Vec<BasicBlock>,
    },
    Resume,
    Return,
    Drop {
        value: Lvalue<'tcx>,
        target: BasicBlock,
        unwind: Option<BasicBlock>,
    },
    Call {
        func: Operand<'tcx>,
        args: Vec<Operand<'tcx>>,
        destination: Option<(Lvalue<'tcx>, BasicBlock)>,
        cleanup: Option<BasicBlock>,
    },
}
