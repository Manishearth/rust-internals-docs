std::os

Module std::os::unix [−] [src]

Experimental extensions to std for Unix platforms.

For now, this module is limited to extracting file descriptors, but its functionality will grow over time.

Example

use std::fs::File; use std::os::unix::prelude::*; fn main() { let f = File::create("foo.txt").unwrap(); let fd = f.as_raw_fd(); // use fd with native unix bindings } 
use std::fs::File;
use std::os::unix::prelude::*;

fn main() {
    let f = File::create("foo.txt").unwrap();
    let fd = f.as_raw_fd();

    // use fd with native unix bindings
}
